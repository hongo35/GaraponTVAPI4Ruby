def initialize(string)
  string.each_line { |line|
    key, val = line.chomp.split(';')
    case key
      when 1 then
        raise "Error: #{val}"
      when 'ipaddr' then
        @ip = val
      when 'gipaddr' then
        @global_ip = val
      when 'pipaddr' then
        @private_ip = val
      when 'port' then
        @port = val
        @global_port = val
      when 'port2' then
        @ts_port = val
      when 'gtvver' then
        @version = val
      when '0' then
        next
      else
        raise "WARNING: unknown response: #{key}  = #{val}"
    end
  }
  if @ip == @private_ip
    @port = 80
  end
end
